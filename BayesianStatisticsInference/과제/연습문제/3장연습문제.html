pos <- function(prior,prob){
  s=0
  for(i in 1:9){
    post <- prior*(i*0.1)^3*(1-i*0.1)^2
    s <- s+post
  }
  p <- (prior*prob^3*(1-prob)^2)/s
  return(p)
}

den <- function(x){
  o <- 0
  for(i in 1:length(x)){
    o[i] <- exp(1)^(-x[i]^2/2)/2/sqrt(2*pi)
  }
  return(o)
}

y1 <- den(x1)
y2 <- den(x2)
